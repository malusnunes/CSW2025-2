def fahrenheit_para_celsius(f):
    """
            Converte temperatura de Fahrenheit para Celsius.
            Fórmula: (F - 32) * 5/9
            """
            celsius = (f - 32) * 5/9
            return celsius
# Exemplo de uso:
temp_f = 98.6
print(f"{temp_f}°F = {fahrenheit_para_celsius(temp_f):.2f}°C")
